27 template <typename Char>
28 class ascii_keyboard
29 {
30  std::unique_ptr<xcb_key_symbols_t, decltype(&xcb_key_symbols_free)> symbols_;
31 
32  // XXX デバッグ用とか備忘録的な意味も兼ねてmapped_type::secondに説明を含めたけど、
33  // 実用的な意味ではmapped_type::firstだけあればいいんじゃねえかな
34  static const std::unordered_map<
35  std::basic_string<Char>, std::pair<Char, std::basic_string<Char>>
36  > control_characters;
37 
38 public:
39  Char code;
40 
41  explicit ascii_keyboard(const std::shared_ptr<xcb_connection_t>& connection)
42  : symbols_ {xcb_key_symbols_alloc(connection.get()), xcb_key_symbols_free}
43  {}
44 
45  // XXX UGLY CODE
46  auto press(const std::shared_ptr<xcb_generic_event_t>& generic_event) noexcept
47  -> bool // is able to output as ASCII code
48  {
49  auto keysym {xcb_key_press_lookup_keysym(
50  symbols_.get(),
51  reinterpret_cast<xcb_key_press_event_t*>(generic_event.get()),
52  reinterpret_cast<xcb_key_press_event_t*>(generic_event.get())->state & ~XCB_MOD_MASK_CONTROL
53  )};
54 
55  if (xcb_is_modifier_key(keysym))
56  {
57  return false;
58  }
59 
60  switch (reinterpret_cast<xcb_key_press_event_t*>(generic_event.get())->state)
61  {
62  case XCB_MOD_MASK_CONTROL:
63  case XCB_MOD_MASK_CONTROL | XCB_MOD_MASK_SHIFT:
64  {
65  const std::string caret {'^', static_cast<Char>(std::toupper(keysym))};
66  if (control_characters.find(caret) != std::end(control_characters))
67  {
68  code = control_characters.at(caret).first;
69  return true;
70  }
71  }
72  break;
73 
74  default:
75  if (0x1F < keysym && keysym < 0x7F)
76  {
77  code = static_cast<Char>(keysym);
78  return true;
79  }
80  break;
81  }
82 
83  return false;
84  }
85 };
